# 문제 링크
[1021 회전하는 큐](https://www.acmicpc.net/problem/1021)

# 아이디어
문제에 제시된 왼쪽으로 이동, 오른쪽으로 이동을 용이하게 하기 위해 덱을 이용해서 풀었다. front에서 찾는 원소까지 거리를 계산하고 그 거리가 가까운 쪽으로 이동하면 된다.

# (1) 처음 접근- 브루트 포스(시간 초과)
front에서 거리를 계산하기 위해 오른쪽에서의 거리, 왼쪽에서의 거리를 브루트 포스로 무모하게 구한 다음 그 값을 비교해서 이동하는 방식으로 구현했는데 예상했듯이 시간 초과가 나왔다. 

# (1) 코드
```c++
#include <iostream>
#include <deque>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    
    int n;
    cin >> n;
 
    deque<int> d;
    
    for(int i = 1 ;i <=n; i++)
        d.push_back(i);
    
    int k;
    cin >> k;
    
    int ans = 0;
    
    for(int i = 0; i < k; i++)
    {
        int num;
        cin >> num;
        
        while (true)
        {
            if(d.front() == num)
            {
                d.pop_front();
                break;
            }
            else
            {
                int l = 0;
                int r = 0;
                
                auto r_i = d.begin();
                auto l_i = d.end();
                
                while (*r_i != num)
                {
                    r_i++;
                    r++;
                }
                while (*l_i != num)
                {
                    l_i--;
                    l++;
                }
                
                if(l < r)
                {
                    ans += l;
                    for(int i = 0; i < l; i++)
                    {
                        int a = d.back();
                        d.pop_back();
                        d.push_front(a);
                    }
                }
                else
                {
                    ans += r;
                    for(int i = 0; i < r; i++)
                    {
                        int a = d.front();
                        d.pop_front();
                        d.push_back(a);
                    }
                }
            }
        }
    }
    
    cout << ans;
}
```
# (2) 두 번째 접근- algorithm의 find() 함수
algorithm의 find 함수를 쓰면 해당하는 값의 이터레이터를 얻어 덱의 주소값을 얻을 수 있기 때문에 거리를 계산하기 용이했다.

# (2) 코드
```c++
#include <iostream>
#include <deque>
#include <algorithm>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int N;
    int M;
    int ans = 0;
    cin>>N>>M;

    deque<int> d;

    for(int i=0;i<N;i++) d.push_back(i + 1);

    for(int i=0;i<M;i++)
    {
        int a;
        cin >> a;
        
        while (true)
        {
            auto index = find(d.begin(),d.end(),a);
        
            if(!d.empty() && *index == d.front())
            {
                d.pop_front();
                break;
            }
            if(index <= d.begin() + d.size()/2)
            {
                int front = d.front();
                d.pop_front();
                d.push_back(front);
                ans++;
            }
            else
            {
                int back = d.back();
                d.pop_back();
                d.push_front(back);
                ans++;
            }
        }
    }

    cout << ans;
    
    return 0;
}
```
